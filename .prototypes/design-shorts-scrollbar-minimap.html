/* A - filmstrip minimap */
.shorts-scrollbar {
	position: relative;
	height: 48px;
	margin-top: 10px;
	cursor: grab;
	overflow: hidden;
}
.shorts-scrollbar__filmstrip {
	position: absolute;
	inset: 0;
	display: flex;
}
.shorts-scrollbar__filmstrip img {
	flex: 1 1 20%;
	height: 100%;
	object-fit: cover;
	display: block;
}
.shorts-scrollbar__dim {
	position: absolute;
	inset: 0;
	background: rgba(15, 23, 42, 0.5);
	/* mask: transparent over the window, black (dim) elsewhere, ~8% soft edge -
	   same buildDimMask idiom as Minimap.tsx, ported to `to right` */
	mask-image: linear-gradient(to right, black 0%, black calc(var(--win-left) - 8%),
		transparent var(--win-left), transparent calc(var(--win-left) + var(--win-width)),
		black calc(var(--win-left) + var(--win-width) + 8%), black 100%);
	-webkit-mask-image: /* same */;
	pointer-events: none;
}
.shorts-scrollbar__thumb {
	position: absolute;
	top: 0; bottom: 0; left: 0;
	width: 60%; /* JS-set: viewportW / contentW, 3 of 5 cards */
	border: 2px solid rgba(15, 23, 42, 0.4);
	background: transparent;
	pointer-events: none;
}
.shorts-scrollbar--night .shorts-scrollbar__dim { background: rgba(255, 255, 255, 0.35); }
.shorts-scrollbar--night .shorts-scrollbar__thumb { border-color: rgba(255, 255, 255, 0.55); }
